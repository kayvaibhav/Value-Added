using namespace std;
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    int t;
    cin>>t;
    while(t-->0)
    {
        int n,k,sand=0,a,b;
        cin>>n>>k;
        sand=n;
        while(n>=k)
        {
            a=n/k;
            b=n%k;
            sand+=a;
            n=a+b;
        }
        cout<<sand<<endl;
    }
    return 0;
}
